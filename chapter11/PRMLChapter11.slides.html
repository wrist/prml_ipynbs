from scipy.special import gamma

def gamma_dist(xs, a=1.0, b=1.0):
    return (b ** a) * (xs ** (a - 1)) * np.exp(- b * xs) / gamma(a)
    
def generalized_cauchy_dist(xs, c=0., b=1., k=1.):
    return k / (1 + ((xs - c) ** 2 / b ** 2))

# [0, 1]
# tan[0, 1] -> [-pi/2, pi/2]
def uniform_to_cauchy(ys, c=0., b=1.):
    return b * np.tan(np.pi*(ys-0.5)) + c
